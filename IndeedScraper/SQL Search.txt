SELECT Jobs.jobTitle, Jobs.salary, Jobs.description, Jobs.date, Jobs.jobLink, Jobs.fullTime, Jobs.permanent, Jobs.remote, Companies.name, Companies.overallRating, Companies.numberOfRating, Companies.compensationRating, Companies.cultureRating, Companies.managementRating, Companies.securityAdvancementRating, Companies.workLifeBalanceRating FROM Jobs JOIN Companies JOIN Keywords ON Jobs.companies_id = Companies.id AND Keywords.jobs_id = Jobs.id WHERE Keywords.Javascript = 0 AND Jobs.stillActive = 1 ORDER BY date DESC
